import struct
import binascii
from Crypto.Util.number import bytes_to_long

img = open('C:/Users/Administrator/Desktop/0.png', 'rb').read()

for i in range(0xFFFF):
    stream = img[12:20] + struct.pack('>i', i) + img[24:29]
    crc = binascii.crc32(stream)
    if crc == bytes_to_long(img[29:33]):
        print(i, hex(i))

70 misc_bbmpp

下载得到加密的zip格式压缩包。尝试后发现不是伪加密，那么再爆破一下：
密码333520，解压得到无扩展名文件bbmppp，十六进制编辑器打开：
由文件名和大量的FF FF FF可以联想到24位色的BMP格式图像，若背景为白色，则数据部分就会出现大量FF FF FF（对应BGR通道255 255 255）。计算可得数据部分的长度为80688 = 0x13b30。将大量FF之前的部分视为文件头，观察其特征：
00 00 00 00 36 00 00 00 28 00 00 00 A4 00 00 00
A4 00 00 00 01 00 18 00 00 00 00 00 00 00 00 00
C4 0E 00 00 C4 0E 00 00 00 00 00 00 00 00 00 00
注意标红部分，按小端序读取，36 00 00 00 = 0x36 = 54，是一般情况下bfOffBits的大小（BITMAPFILEHEADER长度14 + BITMAPINFOHEADER长度40），28 00 00 00 = 0x28 = 40，是一般情况下biSize的大小（即BITMAPINFOHEADER长度40）。A4 00 00 00 = 0xA4 = 164，宽高均为164的情况下像素数据部分长度应为 164 x 164 x 3 = 80688，正好和之前得到的数据部分长度相等。因此可以判定，这是一个缺少了开头部分的BMP格式文件头，将其补全：
42 4D DA 28 00 00 00 00 00 00 36 00 00 00 28 00
00 00 44 00 00 00 33 00 00 00 01 00 18 00 00 00
00 00 A4 28 00 00 00 00 00 00 00 00 00 00 00 00
00 00 00 00 00 00

for i in range(256):
    res = ''
